package com.example.shoppinglist.di

import com.example.shoppinglist.data.local.ShoppingListDataStore
import com.example.shoppinglist.data.repository.SettingsRepositoryImpl
import com.example.shoppinglist.data.repository.ShoppingListRepositoryImpl
import com.example.shoppinglist.domain.repository.SettingsRepository
import com.example.shoppinglist.domain.repository.ShoppingListRepository
import com.google.gson.Gson
import dagger.Module
import dagger.Provides
import dagger.hilt.InstallIn
import dagger.hilt.components.SingletonComponent
import javax.inject.Singleton

@Module
@InstallIn(SingletonComponent::class)
object AppModule {

    @Provides
    @Singleton
    fun provideGson(): Gson = Gson()

    @Provides
    @Singleton
    fun provideShoppingListRepository(
        dataStore: ShoppingListDataStore
    ): ShoppingListRepository = ShoppingListRepositoryImpl(dataStore)

    @Provides
    @Singleton
    fun provideSettingsRepository(
        dataStore: ShoppingListDataStore
    ): SettingsRepository = SettingsRepositoryImpl(dataStore)
}
